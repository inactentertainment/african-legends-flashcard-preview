  function imgUrl(id){
    return "https://drive.google.com/uc?export=view&id=" + encodeURIComponent(id);
  }

  const cards = [];
  for(let k = 0; k < ids.length; k += 2){
    cards.push({ front: imgUrl(ids[k]), back: imgUrl(ids[k + 1]) });
  }

  let i = 0;
  let showingFront = true;

  const cardEl = document.getElementById("alCard");
  const imgEl = document.getElementById("alImg");
  const counterEl = document.getElementById("alCounter");
  const prevBtn = document.getElementById("alPrev");
  const nextBtn = document.getElementById("alNext");
  const flipBtn = document.getElementById("alFlip");

  function render(){
    const c = cards[i];
    imgEl.src = showingFront ? c.front : c.back;
    counterEl.textContent = "Card " + (i + 1) + " of " + cards.length;
    prevBtn.disabled = (i === 0);
    nextBtn.disabled = (i === cards.length - 1);
  }

  function flip(){ showingFront = !showingFront; render(); }
  function next(){ if(i < cards.length - 1){ i += 1; showingFront = true; render(); } }
  function prev(){ if(i > 0){ i -= 1; showingFront = true; render(); } }

  cardEl.addEventListener("click", flip);
  flipBtn.addEventListener("click", flip);
  nextBtn.addEventListener("click", next);
  prevBtn.addEventListener("click", prev);

  cardEl.addEventListener("keydown", (e) => {
    if(e.key === "Enter" || e.key === " "){ e.preventDefault(); flip(); }
    if(e.key === "ArrowRight"){ next(); }
    if(e.key === "ArrowLeft"){ prev(); }
  });

  render();
})();
